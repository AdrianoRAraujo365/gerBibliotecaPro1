-- Extensão para UUID (opcional)
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ── Tabela de Livros ─────────────────────────────────────────
CREATE TABLE IF NOT EXISTS livros (
  id         SERIAL PRIMARY KEY,
  titulo     VARCHAR(200) NOT NULL,
  autor      VARCHAR(150) NOT NULL,
  isbn       VARCHAR(20)  UNIQUE,
  ano        INTEGER,
  genero     VARCHAR(80),
  editora    VARCHAR(120),
  status     VARCHAR(20)  NOT NULL DEFAULT 'Disponível'
             CHECK (status IN ('Disponível','Emprestado')),
  criado_em  TIMESTAMP DEFAULT NOW()
);

-- ── Tabela de Usuários ────────────────────────────────────────
CREATE TABLE IF NOT EXISTS usuarios (
  id         SERIAL PRIMARY KEY,
  nome       VARCHAR(150) NOT NULL,
  email      VARCHAR(150) UNIQUE NOT NULL,
  telefone   VARCHAR(25),
  matricula  VARCHAR(30)  UNIQUE,
  criado_em  TIMESTAMP DEFAULT NOW()
);

-- ── Tabela de Empréstimos ─────────────────────────────────────
CREATE TABLE IF NOT EXISTS emprestimos (
  id               SERIAL PRIMARY KEY,
  livro_id         INTEGER NOT NULL REFERENCES livros(id)   ON DELETE RESTRICT,
  usuario_id       INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE RESTRICT,
  data_emprestimo  TIMESTAMP DEFAULT NOW(),
  data_prevista    TIMESTAMP DEFAULT (NOW() + INTERVAL '14 days'),
  data_devolucao   TIMESTAMP,
  status           VARCHAR(20) NOT NULL DEFAULT 'Ativo'
                   CHECK (status IN ('Ativo','Devolvido','Atrasado')),
  observacao       TEXT
);

-- ── Índices para performance ──────────────────────────────────
CREATE INDEX IF NOT EXISTS idx_livros_status   ON livros(status);
CREATE INDEX IF NOT EXISTS idx_livros_titulo   ON livros(titulo);
CREATE INDEX IF NOT EXISTS idx_emp_status      ON emprestimos(status);
CREATE INDEX IF NOT EXISTS idx_emp_livro       ON emprestimos(livro_id);
CREATE INDEX IF NOT EXISTS idx_emp_usuario     ON emprestimos(usuario_id);

// Conexão com PostgreSQL via Neon
// Pool de conexões para alta performance

require('dotenv').config();
const { Pool } = require('pg');

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  ssl: {
    rejectUnauthorized: false  // Necessário para Neon/SSL
  },
  max: 10,
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 5000,
});

// Teste de conexão
pool.connect((err, client, release) => {
  if (err) {
    console.error('❌ Erro ao conectar PostgreSQL (Neon):', err.message);
    return;
  }
  console.log('✅ PostgreSQL (Neon) conectado com sucesso!');
  release();
});

// Helper para queries
const query = (text, params) => pool.query(text, params);

// Helper para transações
const getClient = () => pool.connect();

module.exports = { pool, query, getClient };

// Classe Livro - Representa um livro do acervo
// ════════════════════════════════════════════════════════════
class Livro {
  constructor({ id, titulo, autor, isbn, ano, genero, editora, status, criado_em }) {
    this.id        = id;
    this.titulo    = titulo;
    this.autor     = autor;
    this.isbn      = isbn    || null;
    this.ano       = ano     || null;
    this.genero    = genero  || null;
    this.editora   = editora || null;
    this.status    = status;
    this.criadoEm  = criado_em;
  }

  /** Verifica se o livro está disponível para empréstimo */
  estaDisponivel() {
    return this.status === 'Disponível';
  }

  /** Retorna representação em string */
  toString() {
    return `[${this.id}] "${this.titulo}" - ${this.autor} | ${this.status}`;
  }

  /** Serializa para JSON da API */
  toJSON() {
    return {
      id:       this.id,
      titulo:   this.titulo,
      autor:    this.autor,
      isbn:     this.isbn,
      ano:      this.ano,
      genero:   this.genero,
      editora:  this.editora,
      status:   this.status,
      criadoEm: this.criadoEm
    };
  }
}

// ════════════════════════════════════════════════════════════
// Classe Usuario - Representa um usuário da biblioteca
// ════════════════════════════════════════════════════════════
class Usuario {
  constructor({ id, nome, email, telefone, matricula, criado_em }) {
    this.id        = id;
    this.nome      = nome;
    this.email     = email;
    this.telefone  = telefone  || null;
    this.matricula = matricula || null;
    this.criadoEm  = criado_em;
  }

  /** Retorna nome formatado */
  getNomeFormatado() {
    return this.nome
      .split(' ')
      .map(p => p.charAt(0).toUpperCase() + p.slice(1).toLowerCase())
      .join(' ');
  }

  toString() {
    return `[${this.id}] ${this.nome} | ${this.email}`;
  }

  toJSON() {
    return {
      id:        this.id,
      nome:      this.nome,
      email:     this.email,
      telefone:  this.telefone,
      matricula: this.matricula,
      criadoEm:  this.criadoEm
    };
  }
}

// ════════════════════════════════════════════════════════════
// Classe Biblioteca - Lógica central de negócios
// ════════════════════════════════════════════════════════════
class Biblioteca {
  constructor(db) {
    this.db = db; // { query, getClient }
  }

  // ── Helpers internos ─────────────────────────────────────

  _criarLivro(row) {
    return new Livro(row);
  }

  _criarUsuario(row) {
    return new Usuario(row);
  }

  // ═══════════════════════════════════════════════════════
  // MÉTODOS DE LIVRO
  // ═══════════════════════════════════════════════════════

  /** Cadastra novo livro no banco */
  async cadastrarLivro({ titulo, autor, isbn, ano, genero, editora }) {
    if (!titulo?.trim()) throw new Error('Título é obrigatório.');
    if (!autor?.trim())  throw new Error('Autor é obrigatório.');

    const { rows } = await this.db.query(
      `INSERT INTO livros (titulo, autor, isbn, ano, genero, editora)
       VALUES ($1, $2, $3, $4, $5, $6)
       RETURNING *`,
      [titulo.trim(), autor.trim(), isbn || null, ano || null, genero || null, editora || null]
    );
    return this._criarLivro(rows[0]);
  }

  /** Lista todos os livros com filtros opcionais */
  async listarLivros({ busca, status, genero } = {}) {
    let sql    = 'SELECT * FROM livros WHERE 1=1';
    const params = [];
    let idx    = 1;

    if (busca) {
      sql += ` AND (titulo ILIKE $${idx} OR autor ILIKE $${idx} OR isbn ILIKE $${idx})`;
      params.push(`%${busca}%`);
      idx++;
    }
    if (status && status !== 'todos') {
      sql += ` AND status = $${idx}`;
      params.push(status);
      idx++;
    }
    if (genero) {
      sql += ` AND genero ILIKE $${idx}`;
      params.push(`%${genero}%`);
      idx++;
    }

    sql += ' ORDER BY titulo';
    const { rows } = await this.db.query(sql, params);
    return rows.map(r => this._criarLivro(r));
  }

  /** Busca livro por ID */
  async buscarLivroPorId(id) {
    const { rows } = await this.db.query(
      'SELECT * FROM livros WHERE id = $1', [id]
    );
    return rows.length ? this._criarLivro(rows[0]) : null;
  }

  /** Atualiza livro */
  async atualizarLivro(id, dados) {
    const livro = await this.buscarLivroPorId(id);
    if (!livro) throw new Error('Livro não encontrado.');

    const { titulo, autor, isbn, ano, genero, editora } = dados;
    const { rows } = await this.db.query(
      `UPDATE livros SET
        titulo  = COALESCE($1, titulo),
        autor   = COALESCE($2, autor),
        isbn    = COALESCE($3, isbn),
        ano     = COALESCE($4, ano),
        genero  = COALESCE($5, genero),
        editora = COALESCE($6, editora)
       WHERE id = $7 RETURNING *`,
      [titulo, autor, isbn, ano, genero, editora, id]
    );
    return this._criarLivro(rows[0]);
  }

  /** Remove livro (somente se disponível) */
  async removerLivro(id) {
    const livro = await this.buscarLivroPorId(id);
    if (!livro) throw new Error('Livro não encontrado.');
    if (!livro.estaDisponivel()) {
      throw new Error('Livro está emprestado e não pode ser removido.');
    }
    await this.db.query('DELETE FROM livros WHERE id = $1', [id]);
    return true;
  }

  // ═══════════════════════════════════════════════════════
  // MÉTODOS DE USUÁRIO
  // ═══════════════════════════════════════════════════════

  async cadastrarUsuario({ nome, email, telefone, matricula }) {
    if (!nome?.trim())  throw new Error('Nome é obrigatório.');
    if (!email?.trim()) throw new Error('Email é obrigatório.');

    const { rows } = await this.db.query(
      `INSERT INTO usuarios (nome, email, telefone, matricula)
       VALUES ($1, $2, $3, $4) RETURNING *`,
      [nome.trim(), email.trim().toLowerCase(), telefone || null, matricula || null]
    );
    return this._criarUsuario(rows[0]);
  }

  async listarUsuarios(busca = '') {
    let sql    = 'SELECT * FROM usuarios WHERE 1=1';
    const params = [];

    if (busca) {
      sql += ' AND (nome ILIKE $1 OR email ILIKE $1 OR matricula ILIKE $1)';
      params.push(`%${busca}%`);
    }
    sql += ' ORDER BY nome';
    const { rows } = await this.db.query(sql, params);
    return rows.map(r => this._criarUsuario(r));
  }

  async buscarUsuarioPorId(id) {
    const { rows } = await this.db.query(
      'SELECT * FROM usuarios WHERE id = $1', [id]
    );
    return rows.length ? this._criarUsuario(rows[0]) : null;
  }

  async atualizarUsuario(id, dados) {
    const usuario = await this.buscarUsuarioPorId(id);
    if (!usuario) throw new Error('Usuário não encontrado.');

    const { nome, email, telefone, matricula } = dados;
    const { rows } = await this.db.query(
      `UPDATE usuarios SET
        nome      = COALESCE($1, nome),
        email     = COALESCE($2, email),
        telefone  = COALESCE($3, telefone),
        matricula = COALESCE($4, matricula)
       WHERE id = $5 RETURNING *`,
      [nome, email, telefone, matricula, id]
    );
    return this._criarUsuario(rows[0]);
  }

  async removerUsuario(id) {
    const usuario = await this.buscarUsuarioPorId(id);
    if (!usuario) throw new Error('Usuário não encontrado.');
    await this.db.query('DELETE FROM usuarios WHERE id = $1', [id]);
    return true;
  }

  // ═══════════════════════════════════════════════════════
  // MÉTODOS DE EMPRÉSTIMO
  // ═══════════════════════════════════════════════════════

  /**
   * Valida se o livro pode ser emprestado.
   * Regra principal: livro Emprestado não pode ser emprestado novamente.
   */
  podeEmprestar(livro) {
    return livro instanceof Livro && livro.estaDisponivel();
  }

  /** Registra empréstimo com transação atômica */
  async registrarEmprestimo(livroId, usuarioId, observacao = '') {
    // Busca e valida entidades
    const livro   = await this.buscarLivroPorId(livroId);
    if (!livro)   throw new Error('Livro não encontrado.');

    const usuario = await this.buscarUsuarioPorId(usuarioId);
    if (!usuario) throw new Error('Usuário não encontrado.');

    // ── Regra de negócio: não emprestar livro já emprestado ──
    if (!this.podeEmprestar(livro)) {
      throw new Error(
        `O livro "${livro.titulo}" já está emprestado. ` +
        `Aguarde a devolução para realizar um novo empréstimo.`
      );
    }

    // Transação atômica: INSERT + UPDATE
    const client = await this.db.getClient();
    try {
      await client.query('BEGIN');

      const { rows } = await client.query(
        `INSERT INTO emprestimos (livro_id, usuario_id, observacao)
         VALUES ($1, $2, $3) RETURNING *`,
        [livroId, usuarioId, observacao || null]
      );

      await client.query(
        `UPDATE livros SET status = 'Emprestado' WHERE id = $1`,
        [livroId]
      );

      await client.query('COMMIT');

      return {
        id:              rows[0].id,
        livro:           livro.titulo,
        autor:           livro.autor,
        usuario:         usuario.nome,
        email:           usuario.email,
        dataEmprestimo:  rows[0].data_emprestimo,
        dataPrevista:    rows[0].data_prevista,
        status:          rows[0].status
      };
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    } finally {
      client.release();
    }
  }

  /** Registra devolução com transação atômica */
  async registrarDevolucao(emprestimoId) {
    const { rows: check } = await this.db.query(
      `SELECT e.*, l.titulo, l.autor, u.nome, u.email
       FROM emprestimos e
       JOIN livros   l ON l.id = e.livro_id
       JOIN usuarios u ON u.id = e.usuario_id
       WHERE e.id = $1 AND e.status = 'Ativo'`,
      [emprestimoId]
    );

    if (!check.length) {
      throw new Error('Empréstimo ativo não encontrado.');
    }
    const emp    = check[0];
    const client = await this.db.getClient();

    try {
      await client.query('BEGIN');

      const { rows } = await client.query(
        `UPDATE emprestimos
         SET data_devolucao = NOW(), status = 'Devolvido'
         WHERE id = $1 RETURNING *`,
        [emprestimoId]
      );

      await client.query(
        `UPDATE livros SET status = 'Disponível' WHERE id = $1`,
        [emp.livro_id]
      );

      await client.query('COMMIT');

      // Verifica atraso
      const atrasado = new Date() > new Date(emp.data_prevista);

      return {
        id:               emprestimoId,
        livro:            emp.titulo,
        autor:            emp.autor,
        usuario:          emp.nome,
        email:            emp.email,
        dataEmprestimo:   emp.data_emprestimo,
        dataDevolucao:    rows[0].data_devolucao,
        atrasado,
        diasAtraso: atrasado
          ? Math.ceil((new Date() - new Date(emp.data_prevista)) / 86400000)
          : 0
      };
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    } finally {
      client.release();
    }
  }

  /** Lista empréstimos com filtros */
  async listarEmprestimos({ status, usuarioId, livroId } = {}) {
    let sql = `
      SELECT
        e.id,
        l.titulo       AS livro,
        l.autor        AS autor,
        u.nome         AS usuario,
        u.email        AS email,
        u.matricula    AS matricula,
        e.data_emprestimo,
        e.data_prevista,
        e.data_devolucao,
        e.status,
        e.observacao,
        CASE
          WHEN e.status = 'Ativo' AND NOW() > e.data_prevista
          THEN TRUE ELSE FALSE
        END AS atrasado
      FROM emprestimos e
      JOIN livros   l ON l.id = e.livro_id
      JOIN usuarios u ON u.id = e.usuario_id
      WHERE 1=1
    `;
    const params = [];
    let idx = 1;

    if (status && status !== 'todos') {
      sql += ` AND e.status = $${idx++}`;
      params.push(status);
    }
    if (usuarioId) {
      sql += ` AND e.usuario_id = $${idx++}`;
      params.push(usuarioId);
    }
    if (livroId) {
      sql += ` AND e.livro_id = $${idx++}`;
      params.push(livroId);
    }

    sql += ' ORDER BY e.data_emprestimo DESC';
    const { rows } = await this.db.query(sql, params);
    return rows;
  }

  // ═══════════════════════════════════════════════════════
  // RELATÓRIOS
  // ═══════════════════════════════════════════════════════

  /** Relatório geral de status de todos os livros */
  async relatorioStatus() {
    const { rows } = await this.db.query(`
      SELECT
        l.id,
        l.titulo,
        l.autor,
        l.isbn,
        l.ano,
        l.genero,
        l.editora,
        l.status,
        u.nome         AS usuario_atual,
        u.email        AS email_usuario,
        u.matricula,
        e.data_emprestimo,
        e.data_prevista,
        CASE
          WHEN e.status = 'Ativo' AND NOW() > e.data_prevista
          THEN TRUE ELSE FALSE
        END AS atrasado
      FROM livros l
      LEFT JOIN emprestimos e ON e.livro_id = l.id AND e.status = 'Ativo'
      LEFT JOIN usuarios    u ON u.id = e.usuario_id
      ORDER BY l.status DESC, l.titulo
    `);
    return rows;
  }

  /** Dashboard: estatísticas gerais */
  async estatisticas() {
    const { rows } = await this.db.query(`
      SELECT
        (SELECT COUNT(*) FROM livros)                                         AS total_livros,
        (SELECT COUNT(*) FROM livros WHERE status = 'Disponível')             AS disponiveis,
        (SELECT COUNT(*) FROM livros WHERE status = 'Emprestado')             AS emprestados,
        (SELECT COUNT(*) FROM usuarios)                                        AS total_usuarios,
        (SELECT COUNT(*) FROM emprestimos WHERE status = 'Ativo')             AS emprestimos_ativos,
        (SELECT COUNT(*) FROM emprestimos WHERE status = 'Devolvido')         AS total_devolvidos,
        (SELECT COUNT(*) FROM emprestimos
          WHERE status = 'Ativo' AND NOW() > data_prevista)                   AS atrasados
    `);
    return rows[0];
  }
}

// ── Helper de resposta de erro ──────────────────────────────
function handleError(res, err) {
  console.error('API Error:', err.message);
  const status = err.message.includes('não encontrado') ? 404
               : err.message.includes('obrigatório')   ? 400
               : err.message.includes('já está')       ? 409
               : 500;

function iniciarMenuConsole() {
  const rl = readline.createInterface({
    input:  process.stdin,
    output: process.stdout
  });

  const ask = (msg) => new Promise(resolve => rl.question(msg, resolve));

  async function menu() {
    console.log('\n╔══════════════════════════════════════════╗');
    console.log('║     MENU - GERENCIAMENTO DE BIBLIOTECA   ║');
    console.log('╠══════════════════════════════════════════╣');
    console.log('║  1. Cadastrar Livro                      ║');
    console.log('║  2. Cadastrar Usuário                    ║');
    console.log('║  3. Registrar Empréstimo                 ║');
    console.log('║  4. Registrar Devolução                  ║');
    console.log('║  5. Listar Livros                        ║');
    console.log('║  6. Listar Usuários                      ║');
    console.log('║  7. Relatório de Status                  ║');
    console.log('║  8. Estatísticas                         ║');
    console.log('║  9. Sair                                 ║');
    console.log('╚══════════════════════════════════════════╝');

    const op = await ask('Opção: ');

    switch (op.trim()) {
      case '1': {
        console.log('\n── Cadastrar Livro ──');
        const titulo  = await ask('Título: ');
        const autor   = await ask('Autor: ');
        const isbn    = await ask('ISBN: ');
        const ano     = await ask('Ano: ');
        const genero  = await ask('Gênero: ');
        const editora = await ask('Editora: ');
        try {
          const l = await biblioteca.cadastrarLivro({ titulo, autor, isbn, ano: parseInt(ano)||null, genero, editora });
          console.log(`\n✅ ${l.toString()}`);
        } catch(e) { console.log(`\n❌ ${e.message}`); }
        break;
      }
      case '2': {
        console.log('\n── Cadastrar Usuário ──');
        const nome      = await ask('Nome: ');
        const email     = await ask('Email: ');
        const telefone  = await ask('Telefone: ');
        const matricula = await ask('Matrícula: ');
        try {
          const u = await biblioteca.cadastrarUsuario({ nome, email, telefone, matricula });
          console.log(`\n✅ ${u.toString()}`);
        } catch(e) { console.log(`\n❌ ${e.message}`); }
        break;
      }
      case '3': {
        console.log('\n── Registrar Empréstimo ──');
        const livros   = await biblioteca.listarLivros({ status: 'Disponível' });
        if (!livros.length) { console.log('ℹ️  Nenhum livro disponível.'); break; }
        console.log('\nLivros Disponíveis:');
        livros.forEach(l => console.log(`  [${l.id}] ${l.titulo} - ${l.autor}`));
        const livroId   = await ask('\nID do Livro: ');
        const usuarios  = await biblioteca.listarUsuarios();
        console.log('\nUsuários:');
        usuarios.forEach(u => console.log(`  [${u.id}] ${u.nome}`));
        const usuarioId = await ask('\nID do Usuário: ');
        const obs       = await ask('Observação (opcional): ');
        try {
          const r = await biblioteca.registrarEmprestimo(parseInt(livroId), parseInt(usuarioId), obs);
          console.log(`\n✅ Empréstimo #${r.id}: "${r.livro}" → ${r.usuario}`);
        } catch(e) { console.log(`\n❌ ${e.message}`); }
        break;
      }
      case '4': {
        console.log('\n── Registrar Devolução ──');
        const emp = await biblioteca.listarEmprestimos({ status: 'Ativo' });
        if (!emp.length) { console.log('ℹ️  Nenhum empréstimo ativo.'); break; }
        console.log('\nEmpréstimos Ativos:');
        emp.forEach(e => console.log(
          `  [${e.id}] "${e.livro}" → ${e.usuario} (desde ${new Date(e.data_emprestimo).toLocaleDateString('pt-BR')})`
        ));
        const id = await ask('\nID do Empréstimo: ');
        try {
          const r = await biblioteca.registrarDevolucao(parseInt(id));
          const msg = r.atrasado ? ` ⚠️ ATRASADO ${r.diasAtraso} dias` : '';
          console.log(`\n✅ Devolução: "${r.livro}"${msg}`);
        } catch(e) { console.log(`\n❌ ${e.message}`); }
        break;
      }
      case '5': {
        const livros = await biblioteca.listarLivros();
        console.log('\n── Lista de Livros ──');
        livros.forEach(l => {
          const ic = l.status === 'Disponível' ? '🟢' : '🔴';
          console.log(`  ${ic} ${l.toString()}`);
        });
        break;
      }
      case '6': {
        const usuarios = await biblioteca.listarUsuarios();
        console.log('\n── Lista de Usuários ──');
        usuarios.forEach(u => console.log(`  👤 ${u.toString()}`));
        break;
      }
      case '7': {
        const dados = await biblioteca.relatorioStatus();
        console.log('\n╔══════════════════════════════════════════════════════════╗');
        console.log('║              RELATÓRIO DE STATUS DOS LIVROS              ║');
        console.log('╠══════════════════════════════════════════════════════════╣');

          const s = d.status === 'Disponível' ? '🟢 DISPONÍVEL' : '🔴 EMPRESTADO';
          const t = d.titulo.padEnd(35).substring(0, 35);
          console.log(`║  ${t} ${s.padEnd(14)} ║`);
          if (d.usuario_atual) {
            console.log(`║  → ${d.usuario_atual.padEnd(52)} ║`);
          }
        });
        console.log('╚══════════════════════════════════════════════════════════╝');
        break;
      }
      case '8': {
        const s = await biblioteca.estatisticas();
        console.log('\n── Estatísticas ──');
        console.log(`  📚 Total Livros:      ${s.total_livros}`);
        console.log(`  🟢 Disponíveis:       ${s.disponiveis}`);
        console.log(`  🔴 Emprestados:       ${s.emprestados}`);
        console.log(`  👤 Usuários:          ${s.total_usuarios}`);
        console.log(`  📋 Empréstimos Ativos:${s.emprestimos_ativos}`);
        console.log(`  ⚠️  Atrasados:         ${s.atrasados}`);
        break;
      }
      case '9': {
        console.log('\n── Relatório Final ──');
        const dados = await biblioteca.relatorioStatus();
        console.log('\nSTATUS FINAL DOS LIVROS:');
        dados.forEach(d => {
          console.log(`  ${d.status === 'Disponível' ? '🟢' : '🔴'} [${d.status.toUpperCase()}] ${d.titulo}`);
        });
        console.log('\n✅ Sistema encerrado. Até logo!\n');
        rl.close();
        return;
      }
      default:
        console.log('⚠️  Opção inválida!');
    }

    await menu();
  }

  menu().catch(console.error);
}

      setLivros(res.dados || []);
    } catch (e) { exibirAlerta(e.message, 'danger'); }
    finally { setLoading(false); }

    const timer = setTimeout(carregar, 400);
    return () => clearTimeout(timer);

  function exibirAlerta(txt, tipo = 'success') {

import React, { useState, useEffect, useCallback } from 'react';
import api from '../services/api';
import { jsPDF } from 'jspdf';
import 'jspdf-autotable';

export default function Relatorio() {
  const [dados,     setDados]     = useState([]);
  const [filtro,    setFiltro]    = useState('todos');
  const [loading,   setLoading]   = useState(false);
  const [gerandoPdf, setGerandoPdf] = useState(false);
  const [atualizado, setAtualizado] = useState('');

  const carregar = useCallback(async () => {
    setLoading(true);
    try {
      const res = await api.get('/relatorio/status');
      setDados(res.dados || []);
      setAtualizado(new Date().toLocaleString('pt-BR'));
    } catch (e) { console.error(e); }
    finally { setLoading(false); }
  }, []);

  useEffect(() => { carregar(); }, [carregar]);

  // Dados filtrados
  const dadosFiltrados = dados.filter(d => {
    if (filtro === 'disponiveis') return d.status === 'Disponível';
    if (filtro === 'emprestados') return d.status === 'Emprestado';
    if (filtro === 'atrasados')   return d.atrasado;
    return true;
  });

  // Estatísticas
  const total      = dados.length;
  const disp       = dados.filter(d => d.status === 'Disponível').length;
  const emp        = dados.filter(d => d.status === 'Emprestado').length;
  const atrasados  = dados.filter(d => d.atrasado).length;
  const percDisp   = total ? Math.round((disp / total) * 100) : 0;

  // ── Geração de PDF com jsPDF ──────────────────────────────
  async function gerarPDF() {
    setGerandoPdf(true);
    try {
      const doc = new jsPDF({ orientation: 'landscape', unit: 'mm', format: 'a4' });
      const W   = doc.internal.pageSize.getWidth();
      const H   = doc.internal.pageSize.getHeight();

      // ── Capa / Cabeçalho ────────────────────────────────
      // Fundo gradiente simulado
      doc.setFillColor(13, 110, 253);
      doc.rect(0, 0, W, 40, 'F');
      doc.setFillColor(102, 16, 242);
      doc.rect(W - 80, 0, 80, 40, 'F');

      // Título
      doc.setTextColor(255, 255, 255);
      doc.setFontSize(22);
      doc.setFont('helvetica', 'bold');
      doc.text('BIBLIOTECAPRO', 14, 17);
      doc.setFontSize(11);
      doc.setFont('helvetica', 'normal');
      doc.text('Relatório Completo de Status do Acervo', 14, 27);
      doc.setFontSize(9);
      doc.text('FATEC SP · Zona Leste · DSM · WEB3', 14, 35);

      // Data / hora no canto
      doc.setFontSize(9);
      doc.text(`Emitido em:`, W - 78, 15, { align: 'left' });
      doc.setFont('helvetica', 'bold');
      doc.text(new Date().toLocaleString('pt-BR'), W - 78, 22, { align: 'left' });
      doc.setFont('helvetica', 'normal');
      doc.text(`Adriano R. de Araujo`, W - 78, 32, { align: 'left' });

      // ── Resumo Estatístico ───────────────────────────────
      doc.setTextColor(0);
      const boxY = 48;
      const boxes = [
        { label: 'Total de Livros',   valor: total,    cor: [13,110,253]  },
        { label: 'Disponíveis',       valor: disp,     cor: [25,135,84]   },
        { label: 'Emprestados',       valor: emp,       cor: [220,53,69]  },
        { label: 'Atrasados',         valor: atrasados, cor: [255,193,7]  },
        { label: '% Disponibilidade', valor: `${percDisp}%`, cor: [108,117,125] },
      ];

      const bw = (W - 28) / boxes.length;
      boxes.forEach((b, i) => {
        const x = 14 + i * bw;
        // Card
        doc.setFillColor(...b.cor);
        doc.roundedRect(x, boxY, bw - 4, 24, 3, 3, 'F');
        // Texto
        doc.setTextColor(255);
        doc.setFontSize(18);
        doc.setFont('helvetica', 'bold');
        doc.text(String(b.valor), x + (bw - 4) / 2, boxY + 14, { align: 'center' });
        doc.setFontSize(7);
        doc.setFont('helvetica', 'normal');
        doc.text(b.label, x + (bw - 4) / 2, boxY + 20, { align: 'center' });
      });

      // ── Tabela ───────────────────────────────────────────
      doc.setTextColor(0);
      doc.setFontSize(11);
      doc.setFont('helvetica', 'bold');
      doc.text('Detalhamento do Acervo', 14, boxY + 32);

      const tableData = dadosFiltrados.map(d => [
        d.id,
        d.titulo.length > 35 ? d.titulo.substring(0, 35) + '...' : d.titulo,
        d.autor,
        d.genero  || '—',
        d.ano     || '—',
        d.status,
        d.usuario_atual || '—',
        d.data_emprestimo
          ? new Date(d.data_emprestimo).toLocaleDateString('pt-BR')
          : '—',
        d.data_prevista
          ? new Date(d.data_prevista).toLocaleDateString('pt-BR')
          : '—',
        d.atrasado ? '⚠️ SIM' : 'Não'
      ]);

      doc.autoTable({
        startY: boxY + 36,
        head: [['ID','Título','Autor','Gênero','Ano','Status','Usuário','Empréstimo','Previsão','Atrasado']],
        body: tableData,
        headStyles: {
          fillColor: [26, 35, 50],
          textColor: [255,255,255],
          fontStyle: 'bold',
          fontSize: 8
        },
        bodyStyles:    { fontSize: 8 },
        alternateRowStyles: { fillColor: [248, 250, 252] },
        columnStyles: {
          0: { cellWidth: 10 },
          5: { fontStyle: 'bold' },
          9: { fontStyle: 'bold' }
        },
        didDrawCell: (data) => {
          if (data.section === 'body' && data.column.index === 5) {
            const status = data.cell.raw;
            if (status === 'Disponível') {
              data.cell.styles.textColor = [25, 135, 84];
            } else {
              data.cell.styles.textColor = [220, 53, 69];
            }
          }
          if (data.section === 'body' && data.column.index === 9) {
            if (data.cell.raw === '⚠️ SIM') {
              data.cell.styles.textColor = [220, 53, 69];
            }
          }
        },
        margin: { left: 14, right: 14 }
      });

      // ── Rodapé de cada página ────────────────────────────
      const totalPags = doc.internal.getNumberOfPages();
      for (let p = 1; p <= totalPags; p++) {
        doc.setPage(p);
        doc.setFillColor(240, 244, 248);
        doc.rect(0, H - 12, W, 12, 'F');
        doc.setFontSize(8);
        doc.setTextColor(100);
        doc.setFont('helvetica', 'normal');
        doc.text('BibliotecaPro © FATEC SP Zona Leste 2025', 14, H - 4);
        doc.text(`Página ${p} de ${totalPags}`, W - 14, H - 4, { align: 'right' });
      }

      // ── Salva o PDF ──────────────────────────────────────
      const dataHoje = new Date().toLocaleDateString('pt-BR').replace(/\//g, '-');
      doc.save(`relatorio-biblioteca-${dataHoje}.pdf`);
    } finally {
      setGerandoPdf(false);
    }
  }

  // ── Comprovante de Devolução ──────────────────────────────
  function gerarComprovante(item) {
    const doc  = new jsPDF({ unit: 'mm', format: 'a5' });
    const W    = doc.internal.pageSize.getWidth();

    // Cabeçalho
    doc.setFillColor(13, 110, 253);
    doc.rect(0, 0, W, 35, 'F');
    doc.setTextColor(255);
    doc.setFontSize(16);
    doc.setFont('helvetica', 'bold');
    doc.text('COMPROVANTE DE DEVOLUÇÃO', W / 2, 14, { align: 'center' });
    doc.setFontSize(9);
    doc.setFont('helvetica', 'normal');
    doc.text('BibliotecaPro · FATEC SP · Zona Leste', W / 2, 22, { align: 'center' });
    doc.text(`Nº Protocolo: EMP-${item.id || '000'}`, W / 2, 30, { align: 'center' });

    // Conteúdo
    doc.setTextColor(0);
    let y = 46;
    const linha = (label, valor) => {
      doc.setFont('helvetica', 'bold');
      doc.setFontSize(9);
      doc.text(label, 12, y);
      doc.setFont('helvetica', 'normal');
      doc.text(String(valor || '—'), 55, y);
      y += 9;
    };

    doc.setFillColor(248, 250, 252);
    doc.roundedRect(10, 38, W - 20, 80, 3, 3, 'F');

    y = 46;
    linha('Livro:',          item.titulo);
    linha('Autor:',          item.autor);
    linha('Usuário:',        item.usuario_atual || '—');
    linha('Matrícula:',      item.matricula     || '—');
    linha('Data Empréstimo:',item.data_emprestimo
      ? new Date(item.data_emprestimo).toLocaleDateString('pt-BR') : '—');
    linha('Previsão Devolução:',item.data_prevista
      ? new Date(item.data_prevista).toLocaleDateString('pt-BR') : '—');
    linha('Data Devolução:',  new Date().toLocaleString('pt-BR'));
    linha('Status:',         item.atrasado ? '⚠️ Devolvido com Atraso' : '✅ Devolvido no Prazo');

    // Linha de assinatura
    y += 10;
    doc.setDrawColor(180);
    doc.setLineWidth(0.5);
    doc.line(12, y, W - 12, y);
    doc.setFontSize(8);
    doc.setTextColor(150);
    doc.text('Assinatura do Bibliotecário', 12, y + 8);
    doc.text('Assinatura do Usuário', W - 12, y + 8, { align: 'right' });

    // Rodapé
    doc.setFillColor(26, 35, 50);
    doc.rect(0, doc.internal.pageSize.getHeight() - 16, W, 16, 'F');
    doc.setTextColor(255);
    doc.setFontSize(8);
    doc.text(
      `Documento gerado em ${new Date().toLocaleString('pt-BR')} · Válido como comprovante oficial.`,
      W / 2,
      doc.internal.pageSize.getHeight() - 6,
      { align: 'center' }
    );

    doc.save(`comprovante-devolucao-${item.id || 'livro'}.pdf`);
  }

  return (

      <div className="row g-3 mb-4">
        {[
          { l: 'Total',        v: total,     b: '#dbeafe', t: '#1e40af', p: '' },
          { l: 'Disponíveis',  v: disp,      b: '#d1fae5', t: '#065f46', p: `${percDisp}%` },
          { l: 'Emprestados',  v: emp,       b: '#fee2e2', t: '#991b1b', p: `${total ? 100-percDisp : 0}%` },
          { l: 'Atrasados',    v: atrasados, b: '#fef3c7', t: '#92400e', p: '' },
        ].map((s, i) => (

                <div style={{ fontSize: '2.2rem', fontWeight: 800, color: s.t }}>{s.v}</div>
                <div className="text-muted small">{s.l}</div>
                {s.p && <div style={{ fontSize: '.75rem', fontWeight: 600, color: s.t }}>{s.p}</div>}
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* Barra de progresso */}
      <div className="card mb-4">
        <div className="card-body py-3">
          <div className="d-flex justify-content-between mb-1">
            <small className="fw-semibold">Taxa de Disponibilidade</small>
            <small className="fw-bold text-primary">{percDisp}%</small>
          </div>

                  </td>
                  <td><small>{d.autor}</small></td>
                  <td>
                    {d.genero && (
                      <span className="badge rounded-pill"
                            style={{ background: '#ede9fe', color: '#5b21b6', fontWeight: 600, fontSize: '.73rem' }}>

                      </span>
                    )}
                  </td>
                  <td><small>{d.ano || '—'}</small></td>
                  <td>
                    <span className={`status-pill ${d.status === 'Disponível' ? 'disponivel' : d.atrasado ? 'atrasado' : 'emprestado'}`}>
                      {d.atrasado ? 'Atrasado' : d.status}
                    </span>
                  </td>
                  <td>
                    {d.usuario_atual
                      ? <div>
                          <div style={{ fontSize: '.88rem', fontWeight: 500 }}>{d.usuario_atual}</div>
                          <div className="text-muted" style={{ fontSize: '.73rem' }}>{d.matricula || d.email_usuario}</div>
                        </div>
                      : <span className="text-muted">—</span>
                    }
                  </td>
                  <td>
                    <small>{d.data_emprestimo
                      ? new Date(d.data_emprestimo).toLocaleDateString('pt-BR')

                  </td>

                      {d.data_prevista
                        ? new Date(d.data_prevista).toLocaleDateString('pt-BR')
                        : '—'}

                    </small>
                  </td>
                  <td>
                    {d.status === 'Emprestado' && (
                      <button
                        className="btn btn-sm"
                        style={{ background: '#fee2e2', color: '#991b1b', borderRadius: '7px', fontSize: '.78rem' }}

                    )}

              ))}
              {!loading && dadosFiltrados.length === 0 && (
                <tr>
                  <td colSpan="10" className="text-center py-5 text-muted">
                    <i className="bi bi-file-earmark-x d-block mb-2" style={{ fontSize: '2rem' }}></i>
                    Nenhum registro encontrado.
                  </td>
                </tr>
              )}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
}

  View, Text, StyleSheet, FlatList, TextInput,

  Modal, ScrollView, Dimensions

import { Ionicons } from '@expo/vector-icons';
import axios from 'axios';

const API   = 'http://192.168.1.X:3001/api';
const FORM0 = { titulo: '', autor: '', isbn: '', ano: '', genero: '', editora: '' };

export default function LivrosScreen() {
  const [livros,    setLivros]    = useState([]);
  const [busca,     setBusca]     = useState('');
  const [statusFiltro, setStatusFiltro] = useState('todos');
  const [loading,   setLoading]   = useState(true);
  const [modal,     setModal]     = useState(false);
  const [form,      setForm]      = useState(FORM0);
  const [editId,    setEditId]    = useState(null);
  const [saving,    setSaving]    = useState(false);

  const carregar = useCallback(async () => {
    setLoading(true);
    try {
      const { data } = await axios.get(`${API}/livros`, {
        params: { busca, status: statusFiltro }
      });
      setLivros(data.dados || []);
    } catch {
      Alert.alert('Erro', 'Não foi possível carregar os livros.');
    } finally { setLoading(false); }
  }, [busca, statusFiltro]);

  useEffect(() => {
    const t = setTimeout(carregar, 500);
    return () => clearTimeout(t);
  }, [carregar]);

  function abrirModal(livro = null) {
    setForm(livro
      ? { titulo: livro.titulo, autor: livro.autor, isbn: livro.isbn || '',
          ano: String(livro.ano || ''), genero: livro.genero || '', editora: livro.editora || '' }
      : FORM0);
    setEditId(livro?.id || null);
    setModal(true);
  }

  async function salvar() {
    if (!form.titulo || !form.autor) {
      return Alert.alert('Atenção', 'Título e autor são obrigatórios.');
    }
    setSaving(true);
    try {
      if (editId) {
        await axios.put(`${API}/livros/${editId}`, form);
        Alert.alert('✅', 'Livro atualizado!');
      } else {
        await axios.post(`${API}/livros`, { ...form, ano: parseInt(form.ano) || null });
        Alert.alert('✅', 'Livro cadastrado!');
      }
      setModal(false);
      carregar();
    } catch (e) {
      Alert.alert('Erro', e.response?.data?.erro || e.message);
    } finally { setSaving(false); }
  }

  async function deletar(id, titulo) {
    Alert.alert('Confirmar', `Remover "${titulo}"?`, [
      { text: 'Cancelar', style: 'cancel' },
      {
        text: 'Remover', style: 'destructive',
        onPress: async () => {
          try {
            await axios.delete(`${API}/livros/${id}`);
            Alert.alert('✅', 'Livro removido!');
            carregar();
          } catch (e) {
            Alert.alert('Erro', e.response?.data?.erro || e.message);
          }
        }
      }
    ]);
  }

  function renderLivro({ item: l }) {

    return (
      <View style={[styles.card, { borderLeftColor: disp ? '#198754' : '#dc3545' }]}>

            <Text style={styles.cardAutor}>✍️ {l.autor}</Text>
          </View>

            { backgroundColor: disp ? '#d1fae5' : '#fee2e2' }]}>

            <Text style={[styles.statusTxt, { color: disp ? '#065f46' : '#991b1b' }]}>
              {l.status}
            </Text>
          </View>
        </View>
        <View style={styles.cardMeta}>
          {l.genero  && <Text style={styles.metaChip}>🏷️ {l.genero}</Text>}
          {l.ano     && <Text style={styles.metaChip}>📅 {l.ano}</Text>}
          {l.editora && <Text style={styles.metaChip}>🏢 {l.editora}</Text>}
        </View>
        <View style={styles.cardActions}>
          <TouchableOpacity style={styles.btnEditar} onPress={() => abrirModal(l)}>
            <Ionicons name="create-outline" size={16} color="#1d4ed8" />
            <Text style={styles.btnEditarTxt}>Editar</Text>
          </TouchableOpacity>
          <TouchableOpacity
            style={[styles.btnDeletar, { opacity: disp ? 1 : .4 }]}
            onPress={() => disp && deletar(l.id, l.titulo)}
            disabled={!disp}>
            <Ionicons name="trash-outline" size={16} color="#dc2626" />
            <Text style={styles.btnDeletarTxt}>Remover</Text>
          </TouchableOpacity>
        </View>
      </View>
    );

      </Modal>
    </View>
  );
