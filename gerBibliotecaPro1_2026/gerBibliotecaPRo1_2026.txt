-- ============================================
-- Sistema de Gerenciamento de Biblioteca
-- FATEC SP - Zona Leste - DSM - WEB3
-- ============================================

CREATE DATABASE IF NOT EXISTS biblioteca_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE biblioteca_db;

-- Tabela de Livros
CREATE TABLE IF NOT EXISTS livros (
  id      INT AUTO_INCREMENT PRIMARY KEY,
  titulo  VARCHAR(150) NOT NULL,
  autor   VARCHAR(100) NOT NULL,
  isbn    VARCHAR(20)  UNIQUE,
  ano     INT,
  status  ENUM('Disponível','Emprestado') DEFAULT 'Disponível',
  criado_em DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Tabela de Usuários
CREATE TABLE IF NOT EXISTS usuarios (
  id        INT AUTO_INCREMENT PRIMARY KEY,
  nome      VARCHAR(100) NOT NULL,
  email     VARCHAR(100) UNIQUE NOT NULL,
  telefone  VARCHAR(20),
  criado_em DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Tabela de Empréstimos
CREATE TABLE IF NOT EXISTS emprestimos (
  id               INT AUTO_INCREMENT PRIMARY KEY,
  livro_id         INT NOT NULL,
  usuario_id       INT NOT NULL,
  data_emprestimo  DATETIME DEFAULT CURRENT_TIMESTAMP,
  data_devolucao   DATETIME NULL,
  status           ENUM('Ativo','Devolvido') DEFAULT 'Ativo',
  FOREIGN KEY (livro_id)   REFERENCES livros(id)   ON DELETE RESTRICT,
  FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE RESTRICT
);

// ============================================
// POO - Classes: Livro, Usuario, Biblioteca
// FATEC SP - Zona Leste - WEB3
// ============================================

const pool = require('../database/connection');

// ─── Classe Livro ────────────────────────────
class Livro {
  constructor(id, titulo, autor, isbn, ano, status) {
    this.id     = id;
    this.titulo = titulo;
    this.autor  = autor;
    this.isbn   = isbn;
    this.ano    = ano;
    this.status = status;
  }

  /** Verifica se o livro está disponível para empréstimo */
  estaDisponivel() {
    return this.status === 'Disponível';
  }

  /** Retorna representação em string */
  toString() {
    return `[${this.id}] ${this.titulo} - ${this.autor} (${this.status})`;
  }
}

// ─── Classe Usuario ───────────────────────────
class Usuario {
  constructor(id, nome, email, telefone) {
    this.id       = id;
    this.nome     = nome;
    this.email    = email;
    this.telefone = telefone;
  }

  /** Retorna representação em string */
  toString() {
    return `[${this.id}] ${this.nome} <${this.email}>`;
  }
}

// ─── Classe Biblioteca (lógica central) ───────
class Biblioteca {
  constructor(db) {
    this.db = db; // pool de conexão MySQL
  }

  // ══════════════════════════════════════
  // MÉTODOS DE LIVRO
  // ══════════════════════════════════════

  /** Cadastra um novo livro no banco */
  async cadastrarLivro({ titulo, autor, isbn, ano }) {
    if (!titulo || !autor) {
      throw new Error('Título e autor são obrigatórios.');
    }

    const [result] = await this.db.execute(
      `INSERT INTO livros (titulo, autor, isbn, ano) VALUES (?, ?, ?, ?)`,
      [titulo, autor, isbn || null, ano || null]
    );

    return new Livro(result.insertId, titulo, autor, isbn, ano, 'Disponível');
  }

  /** Lista todos os livros */
  async listarLivros() {
    const [rows] = await this.db.execute(
      `SELECT * FROM livros ORDER BY titulo`
    );
    return rows.map(r => new Livro(r.id, r.titulo, r.autor, r.isbn, r.ano, r.status));
  }

  /** Busca livro por ID */
  async buscarLivroPorId(id) {
    const [rows] = await this.db.execute(
      `SELECT * FROM livros WHERE id = ?`, [id]
    );
    if (rows.length === 0) return null;
    const r = rows[0];
    return new Livro(r.id, r.titulo, r.autor, r.isbn, r.ano, r.status);
  }

  /** Atualiza livro */
  async atualizarLivro(id, dados) {
    const livro = await this.buscarLivroPorId(id);
    if (!livro) throw new Error('Livro não encontrado.');

    const { titulo, autor, isbn, ano } = dados;
    await this.db.execute(
      `UPDATE livros SET titulo=?, autor=?, isbn=?, ano=? WHERE id=?`,
      [
        titulo || livro.titulo,
        autor  || livro.autor,
        isbn   || livro.isbn,
        ano    || livro.ano,
        id
      ]
    );
    return await this.buscarLivroPorId(id);
  }

  /** Remove livro */
  async removerLivro(id) {
    const livro = await this.buscarLivroPorId(id);
    if (!livro) throw new Error('Livro não encontrado.');
    if (!livro.estaDisponivel()) {
      throw new Error('Não é possível remover um livro emprestado.');
    }
    await this.db.execute(`DELETE FROM livros WHERE id=?`, [id]);
    return true;
  }

  // ══════════════════════════════════════
  // MÉTODOS DE USUÁRIO
  // ══════════════════════════════════════

  /** Cadastra usuário */
  async cadastrarUsuario({ nome, email, telefone }) {
    if (!nome || !email) {
      throw new Error('Nome e email são obrigatórios.');
    }
    const [result] = await this.db.execute(
      `INSERT INTO usuarios (nome, email, telefone) VALUES (?, ?, ?)`,
      [nome, email, telefone || null]
    );
    return new Usuario(result.insertId, nome, email, telefone);
  }

  /** Lista usuários */
  async listarUsuarios() {
    const [rows] = await this.db.execute(
      `SELECT * FROM usuarios ORDER BY nome`
    );
    return rows.map(r => new Usuario(r.id, r.nome, r.email, r.telefone));
  }

  /** Busca usuário por ID */
  async buscarUsuarioPorId(id) {
    const [rows] = await this.db.execute(
      `SELECT * FROM usuarios WHERE id = ?`, [id]
    );
    if (rows.length === 0) return null;
    const r = rows[0];
    return new Usuario(r.id, r.nome, r.email, r.telefone);
  }

  /** Remove usuário */
  async removerUsuario(id) {
    const usuario = await this.buscarUsuarioPorId(id);
    if (!usuario) throw new Error('Usuário não encontrado.');
    await this.db.execute(`DELETE FROM usuarios WHERE id=?`, [id]);
    return true;
  }

  // ══════════════════════════════════════
  // MÉTODOS DE EMPRÉSTIMO - Regra (B)
  // ══════════════════════════════════════

  /**
   * Valida se o livro pode ser emprestado.
   * Regra (B): livro já emprestado não pode ser emprestado novamente.
   */
  podeEmprestar(livro) {
    return livro.estaDisponivel();
  }

  /** Registra empréstimo com validação POO */
  async registrarEmprestimo(livroId, usuarioId) {
    // Valida livro
    const livro = await this.buscarLivroPorId(livroId);
    if (!livro) throw new Error('Livro não encontrado.');

    // Valida usuário
    const usuario = await this.buscarUsuarioPorId(usuarioId);
    if (!usuario) throw new Error('Usuário não encontrado.');

    // Regra (B): verifica disponibilidade via método POO
    if (!this.podeEmprestar(livro)) {
      throw new Error(
        `O livro "${livro.titulo}" já está emprestado e não pode ser emprestado novamente.`
      );
    }

    // Transação: INSERT emprestimo + UPDATE status livro
    const conn = await this.db.getConnection();
    try {
      await conn.beginTransaction();

      const [result] = await conn.execute(
        `INSERT INTO emprestimos (livro_id, usuario_id) VALUES (?, ?)`,
        [livroId, usuarioId]
      );

      await conn.execute(
        `UPDATE livros SET status='Emprestado' WHERE id=?`,
        [livroId]
      );

      await conn.commit();
      return {
        id:             result.insertId,
        livro:          livro.titulo,
        usuario:        usuario.nome,
        dataEmprestimo: new Date().toISOString()
      };
    } catch (err) {
      await conn.rollback();
      throw err;
    } finally {
      conn.release();
    }
  }

  /** Registra devolução */
  async registrarDevolucao(emprestimoId) {
    const [rows] = await this.db.execute(
      `SELECT e.*, l.titulo FROM emprestimos e
       JOIN livros l ON l.id = e.livro_id
       WHERE e.id = ? AND e.status = 'Ativo'`,
      [emprestimoId]
    );

    if (rows.length === 0) {
      throw new Error('Empréstimo ativo não encontrado.');
    }

    const emprestimo = rows[0];
    const conn = await this.db.getConnection();
    try {
      await conn.beginTransaction();

      await conn.execute(
        `UPDATE emprestimos
         SET data_devolucao=NOW(), status='Devolvido'
         WHERE id=?`,
        [emprestimoId]
      );

      await conn.execute(
        `UPDATE livros SET status='Disponível' WHERE id=?`,
        [emprestimo.livro_id]
      );

      await conn.commit();

      return {
        emprestimoId,
        livro:         emprestimo.titulo,
        dataDevolucao: new Date().toISOString()
      };
    } catch (err) {
      await conn.rollback();
      throw err;
    } finally {
      conn.release();
    }
  }

  /** Lista empréstimos com JOIN */
  async listarEmprestimos() {
    const [rows] = await this.db.execute(`
      SELECT
        e.id,
        l.titulo        AS livro,
        l.autor         AS autor,
        u.nome          AS usuario,
        u.email         AS email,
        e.data_emprestimo,
        e.data_devolucao,
        e.status
      FROM emprestimos e
      JOIN livros   l ON l.id = e.livro_id
      JOIN usuarios u ON u.id = e.usuario_id
      ORDER BY e.data_emprestimo DESC
    `);
    return rows;
  }

  // ══════════════════════════════════════
  // RELATÓRIO - Requisito (C)
  // ══════════════════════════════════════

  /** Relatório completo: todos os livros com status */
  async relatorioStatus() {
    const [rows] = await this.db.execute(`
      SELECT
        l.id,
        l.titulo,
        l.autor,
        l.status,
        u.nome  AS usuario_atual,
        e.data_emprestimo
      FROM livros l
      LEFT JOIN emprestimos e ON e.livro_id = l.id AND e.status = 'Ativo'
      LEFT JOIN usuarios    u ON u.id = e.usuario_id
      ORDER BY l.status DESC, l.titulo
    `);
    return rows;
  }

  /** Relatório: somente livros emprestados no momento */
  async livrosEmprestadosAgora() {
    const [rows] = await this.db.execute(`
      SELECT
        l.titulo,
        l.autor,
        u.nome          AS usuario,
        u.email,
        e.data_emprestimo
      FROM emprestimos e
      JOIN livros   l ON l.id = e.livro_id
      JOIN usuarios u ON u.id = e.usuario_id
      WHERE e.status = 'Ativo'
      ORDER BY e.data_emprestimo
    `);
    return rows;
  }
}

// Exporta instância única (Singleton)
const db = require('../database/connection');
module.exports = new Biblioteca(db);

// ============================================

const express    = require('express');
const router     = express.Router();
const biblioteca = require('../controllers/BibliotecaController');

// ─── LIVROS ──────────────────────────────────

/**
 * @route GET /api/livros
 * @desc  Lista todos os livros
 */
router.get('/livros', async (req, res) => {
  try {
    const livros = await biblioteca.listarLivros();
    res.json(livros);
  } catch (err) {
    res.status(500).json({ erro: err.message });
  }
});

/**
 * @route GET /api/livros/:id
 * @desc  Busca livro por ID
 */
router.get('/livros/:id', async (req, res) => {
  try {
    const livro = await biblioteca.buscarLivroPorId(req.params.id);
    if (!livro) return res.status(404).json({ erro: 'Livro não encontrado.' });
    res.json(livro);
  } catch (err) {
    res.status(500).json({ erro: err.message });
  }
});

/**
 * @route POST /api/livros
 * @desc  Cadastra novo livro
 */
router.post('/livros', async (req, res) => {
  try {
    const livro = await biblioteca.cadastrarLivro(req.body);
    res.status(201).json({ mensagem: 'Livro cadastrado!', livro });
  } catch (err) {
    res.status(400).json({ erro: err.message });
  }
});

/**
 * @route PUT /api/livros/:id
 * @desc  Atualiza livro
 */
router.put('/livros/:id', async (req, res) => {
  try {
    const livro = await biblioteca.atualizarLivro(req.params.id, req.body);
    res.json({ mensagem: 'Livro atualizado!', livro });
  } catch (err) {
    res.status(400).json({ erro: err.message });
  }
});

 * @route DELETE /api/livros/:id

  try {
    await biblioteca.removerLivro(req.params.id);
    res.json({ mensagem: 'Livro removido com sucesso!' });
  } catch (err) {
    res.status(400).json({ erro: err.message });
  }
});

import React, { useState, useEffect } from 'react';
import axios from 'axios';

const API = 'http://localhost:3001/api';

export default function Livros() {
  const [livros,    setLivros]    = useState([]);
  const [form,      setForm]      = useState({ titulo: '', autor: '', isbn: '', ano: '' });
  const [editando,  setEditando]  = useState(null);
  const [busca,     setBusca]     = useState('');
  const [loading,   setLoading]   = useState(false);
  const [mensagem,  setMensagem]  = useState(null);

  useEffect(() => { carregarLivros(); }, []);

  async function carregarLivros() {
    setLoading(true);
    try {
      const { data } = await axios.get(`${API}/livros`);
      setLivros(data);
    } catch {
      exibirMensagem('Erro ao carregar livros.', 'danger');
    } finally {
      setLoading(false);
    }
  }

  function exibirMensagem(texto, tipo = 'success') {
    setMensagem({ texto, tipo });
    setTimeout(() => setMensagem(null), 4000);
  }

  async function handleSubmit(e) {
    e.preventDefault();
    try {
      if (editando) {
        await axios.put(`${API}/livros/${editando}`, form);
        exibirMensagem('Livro atualizado com sucesso!');
        setEditando(null);
      } else {
        await axios.post(`${API}/livros`, form);
        exibirMensagem('Livro cadastrado com sucesso!');
      }
      setForm({ titulo: '', autor: '', isbn: '', ano: '' });
      carregarLivros();
    } catch (err) {
      exibirMensagem(err.response?.data?.erro || 'Erro ao salvar livro.', 'danger');

import { jsPDF } from 'jspdf';
import 'jspdf-autotable';

const API = 'http://localhost:3001/api';

// ─── Componente Relatorio - Requisito (C) ─────────────────────────
export default function Relatorio() {
  const [livros,    setLivros]    = useState([]);
  const [loading,   setLoading]   = useState(false);
  const [filtro,    setFiltro]    = useState('todos');
  const [atualizado, setAtualizado] = useState('');

  useEffect(() => {
    carregarRelatorio();
  }, []);

  async function carregarRelatorio() {
    setLoading(true);
    try {
      const { data } = await axios.get(`${API}/relatorio/status`);
      setLivros(data);
      setAtualizado(new Date().toLocaleString('pt-BR'));
    } catch {
      console.error('Erro ao carregar relatório');
    } finally {
      setLoading(false);
    }
  }

  // ─── Gerar PDF com jsPDF ──────────────────
  function gerarPDF() {
    const doc = new jsPDF();

    // Cabeçalho
    doc.setFillColor(13, 110, 253);
    doc.rect(0, 0, 210, 30, 'F');
    doc.setTextColor(255, 255, 255);
    doc.setFontSize(18);
    doc.text('BibliotecaPro - Relatório de Status', 14, 18);

    doc.setTextColor(0, 0, 0);
    doc.setFontSize(10);
    doc.text(`Gerado em: ${new Date().toLocaleString('pt-BR')}`, 14, 38);
    doc.text(`FATEC SP - Zona Leste - Desenvolvimento de Software Multiplataforma`, 14, 44);

    // Estatísticas
    const disponiveis = livros.filter(l => l.status === 'Disponível').length;
    const emprestados = livros.filter(l => l.status === 'Emprestado').length;

    doc.setFillColor(240, 248, 255);
    doc.rect(14, 50, 55, 18, 'F');
    doc.setFontSize(12);
    doc.text(`Total: ${livros.length}`, 22, 60);

    doc.setFillColor(212, 237, 218);
    doc.rect(74, 50, 55, 18, 'F');
    doc.setTextColor(25, 135, 84);
    doc.text(`Disponíveis: ${disponiveis}`, 82, 60);

    doc.setFillColor(248, 215, 218);
    doc.rect(134, 50, 60, 18, 'F');
    doc.setTextColor(220, 53, 69);
    doc.text(`Emprestados: ${emprestados}`, 142, 60);
    doc.setTextColor(0, 0, 0);

    // Tabela principal
    const dados = livrosFiltrados.map(l => [
      l.id,
      l.titulo,
      l.autor,
      l.status,
      l.usuario_atual || '—',
      l.data_emprestimo
        ? new Date(l.data_emprestimo).toLocaleDateString('pt-BR')
        : '—'
    ]);

    doc.autoTable({
      startY: 76,
      head: [['ID', 'Título', 'Autor', 'Status', 'Usuário', 'Data Empréstimo']],
      body: dados,
      headStyles: { fillColor: [13, 110, 253] },
      alternateRowStyles: { fillColor: [248, 249, 250] },
      didDrawCell: (data) => {
        if (data.column.index === 3 && data.section === 'body') {
          const status = data.cell.raw;
          if (status === 'Disponível') {
            data.cell.styles.textColor = [25, 135, 84];
            data.cell.styles.fontStyle = 'bold';
          } else {
            data.cell.styles.textColor = [220, 53, 69];
            data.cell.styles.fontStyle = 'bold';
          }
        }
      }
    });

    // Rodapé
    const totalPaginas = doc.internal.getNumberOfPages();
    for (let i = 1; i <= totalPaginas; i++) {
      doc.setPage(i);
      doc.setFontSize(8);
      doc.setTextColor(150);
      doc.text(
        `Página ${i} de ${totalPaginas} - BibliotecaPro © FATEC SP 2025`,

      setUsuarios(usrRes.data);
    } catch {
      Alert.alert('Erro', 'Não foi possível carregar os dados.');
    } finally {
      setLoading(false);
    }
  }

  async function registrarEmprestimo() {
    if (!livroId || !usuarioId) {
      Alert.alert('Atenção', 'Selecione o livro e o usuário.');
      return;
    }
    try {
      const res = await axios.post(`${API}/emprestimos`, {
        livro_id:   parseInt(livroId),
        usuario_id: parseInt(usuarioId)
      });
      Alert.alert('✅ Sucesso', `Empréstimo registrado!\n${res.data.livro} → ${res.data.usuario}`);
      setModal(false);
      setLivroId('');
      setUsuarioId('');
      carregarDados();
    } catch (err) {
      // Regra (B): livro já emprestado
      Alert.alert('❌ Erro', err.response?.data?.erro || 'Erro ao registrar empréstimo.');
    }
  }

  async function registrarDevolucao(emprestimo) {
    Alert.alert(
      'Confirmar Devolução',
      `Devolver "${emprestimo.livro}" de ${emprestimo.usuario}?`,
      [
        { text: 'Cancelar', style: 'cancel' },
        {
          text: 'Confirmar',
          style: 'default',
          onPress: async () => {
            try {
              await axios.put(`${API}/emprestimos/${emprestimo.id}/devolver`);
              Alert.alert('✅ Devolução registrada!');
              carregarDados();
            } catch (err) {
              Alert.alert('Erro', err.response?.data?.erro || 'Erro ao devolver.');
            }
          }
        }
      ]
    );
  }

  function renderItem({ item }) {
    const ativo = item.status === 'Ativo';
    return (
      <View style={[styles.card, {
        borderLeftColor: ativo ? '#dc3545' : '#198754'
      }]}>
        <View style={styles.cardHeader}>
          <Text style={styles.livroTitulo} numberOfLines={1}>{item.livro}</Text>
          <View style={[styles.badge, {
            backgroundColor: ativo ? '#f8d7da' : '#d1e7dd'
          }]}>
            <Text style={[styles.badgeText, {
              color: ativo ? '#58151c' : '#0a3622'
            }]}>
              {ativo ? '🔴' : '🟢'} {item.status}
            </Text>
          </View>
        </View>
        <Text style={styles.meta}>👤 {item.usuario}</Text>
        <Text style={styles.meta}>
          📅 {new Date(item.data_emprestimo).toLocaleDateString('pt-BR')}
        </Text>
        {item.data_devolucao && (
          <Text style={styles.meta}>
            ↩️ {new Date(item.data_devolucao).toLocaleDateString('pt-BR')}
          </Text>
        )}
        {ativo && (
          <TouchableOpacity style={styles.devolverBtn}
                            onPress={() => registrarDevolucao(item)}>
            <Text style={styles.devolverBtnText}>↩️ Registrar Devolução</Text>
          </TouchableOpacity>
        )}
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <View style={styles.topBar}>
        <Text style={styles.topBarText}>
          {emprestimos.filter(e => e.status === 'Ativo').length} empréstimos ativos
        </Text>
        <TouchableOpacity style={styles.addBtn} onPress={() => setModal(true)}>
          <Text style={styles.addBtnText}>+ Novo</Text>
        </TouchableOpacity>
      </View>

      {loading ? (
        <View style={styles.center}>
          <ActivityIndicator size="large" color="#0d6efd" />
        </View>
      ) : (
        <FlatList
          data={emprestimos}
          keyExtractor={item => item.id.toString()}
          renderItem={renderItem}
          contentContainerStyle={{ padding: 16 }}
          ListEmptyComponent={
            <View style={styles.center}>
              <Text style={styles.emptyText}>Nenhum empréstimo registrado.</Text>
            </View>
          }
        />
      )}

      {/* Modal Novo Empréstimo */}
      <Modal visible={modal} animationType="slide" transparent>
        <View style={styles.modalOverlay}>
          <View style={styles.modalContent}>
            <Text style={styles.modalTitle}>📋 Novo Empréstimo</Text>

            <Text style={styles.inputLabel}>Livro Disponível *</Text>
            <View style={styles.pickerContainer}>
              <Picker selectedValue={livroId}
                      onValueChange={setLivroId}>
                <Picker.Item label="Selecione um livro..." value="" />
                {livros.map(l => (
                  <Picker.Item key={l.id} label={`${l.titulo} (${l.autor})`} value={l.id.toString()} />
                ))}
              </Picker>
            </View>
            {livros.length === 0 && (
              <Text style={styles.aviso}>⚠️ Nenhum livro disponível no momento.</Text>
            )}

            <Text style={styles.inputLabel}>Usuário *</Text>
            <View style={styles.pickerContainer}>
              <Picker selectedValue={usuarioId}
                      onValueChange={setUsuarioId}>
                <Picker.Item label="Selecione um usuário..." value="" />
                {usuarios.map(u => (
                  <Picker.Item key={u.id} label={`${u.nome} (${u.email})`} value={u.id.toString()} />
                ))}
              </Picker>
            </View>

            <View style={styles.modalBtns}>
              <TouchableOpacity style={styles.btnCancelar}
                                onPress={() => setModal(false)}>
                <Text style={styles.btnCancelarText}>Cancelar</Text>
              </TouchableOpacity>
              <TouchableOpacity
                style={[styles.btnSalvar, { opacity: livros.length === 0 ? 0.5 : 1 }]}

              </TouchableOpacity>
            </View>
          </View>
        </View>
      </Modal>
    </View>
  );
}

class Livro {
  constructor(id, titulo, autor, status = 'Disponível') {
    this.id = id;
    this.titulo = titulo;
    this.autor = autor;
    this.status = status;
  }
}

class Usuario {
  constructor(id, nome, email) {
    this.id = id;
    this.nome = nome;
    this.email = email;
  }
}

class Biblioteca {
  constructor(db) {
    this.db = db;
  }

  // Regra (B): não permite empréstimo se já estiver emprestado
  async podeEmprestar(livroId) {
    const [rows] = await this.db.query(
      "SELECT status FROM livros WHERE id = ?", [livroId]
    );
    return rows[0] && rows[0].status === 'Disponível';
  }

  async cadastrarLivro(titulo, autor) {
    const [result] = await this.db.query(
      "INSERT INTO livros (titulo, autor) VALUES (?, ?)", 
      [titulo, autor]
    );
    return new Livro(result.insertId, titulo, autor);
  }

  async registrarEmprestimo(livroId, usuarioId) {
    const pode = await this.podeEmprestar(livroId);
    if (!pode) throw new Error('Livro já está emprestado');

    await this.db.query(
      `INSERT INTO emprestimos (livro_id, usuario_id, data_emprestimo) 
       VALUES (?, ?, NOW())`,
      [livroId, usuarioId]
    );
    await this.db.query(
      "UPDATE livros SET status = 'Emprestado' WHERE id = ?", 
      [livroId]
    );
    return true;
  }

  async registrarDevolucao(livroId) {
    await this.db.query(
      `UPDATE emprestimos SET data_devolucao = NOW() 
       WHERE livro_id = ? AND data_devolucao IS NULL`,
      [livroId]
    );
    await this.db.query(
      "UPDATE livros SET status = 'Disponível' WHERE id = ?", 
      [livroId]
    );
  }

  async relatorioStatus() {
    const [livros] = await this.db.query(
      "SELECT id, titulo, autor, status FROM livros ORDER BY id"
    );
    return livros;
  }
}

module.exports = { Livro, Usuario, Biblioteca };

const readline = require('readline');
const mysql = require('mysql2/promise');
const { Biblioteca } = require('./controllers/BibliotecaController');

async function main() {
  const connection = await mysql.createConnection({
    host: 'localhost',
    user: 'root',
    password: '',
    database: 'biblioteca_db'
  });

  const biblioteca = new Biblioteca(connection);
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  });

  function menu() {
    console.log('\n=== Gerenciamento de Biblioteca ===');
    console.log('1. Cadastrar Livro');
    console.log('2. Registrar Empréstimo');
    console.log('3. Registrar Devolução');
    console.log('4. Relatório de Status');
    console.log('5. Sair');
    rl.question('\nEscolha uma opção: ', async (opcao) => {
      try {
        if (opcao === '1') {
          rl.question('Título: ', (titulo) => {
            rl.question('Autor: ', async (autor) => {
              await biblioteca.cadastrarLivro(titulo, autor);
              console.log('Livro cadastrado com sucesso!');
              menu();
            });
          });
        } else if (opcao === '2') {
          rl.question('ID do Livro: ', (livroId) => {
            rl.question('ID do Usuário: ', async (usuarioId) => {
              try {
                await biblioteca.registrarEmprestimo(livroId, usuarioId);
                console.log('Empréstimo realizado!');
              } catch (e) {
                console.log(e.message);
              }
              menu();
            });
          });
        } else if (opcao === '3') {
          rl.question('ID do Livro: ', async (livroId) => {
            await biblioteca.registrarDevolucao(livroId);
            console.log('Devolução registrada!');
            menu();
          });
        } else if (opcao === '4') {
          const relatorio = await biblioteca.relatorioStatus();
          console.table(relatorio);
          menu();
        } else if (opcao === '5') {
          rl.close();
          connection.end();
        }
      } catch (err) {
        console.error(err.message);
        menu();
      }
    });
  }
  menu();
}
